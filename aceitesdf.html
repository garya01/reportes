Analisis del IPC de Aceites y los Commodities

1. Introduccin

Este documento presenta un anlisis comparativo entre los ndices de precios de los Commodities y el IPC de aceites, incluyendo:

  • Grficos de tendencias.
  • Calculo de correlaciones.
  • Anlisis de causalidad de Granger.
  • Evaluacin de rezagos (lag).

ndices de Commodities e IPC

# Crear ndices de commodities con base 100 (solo entre 2020-2024)
commodities <- commodities %>%
  mutate(across(COCONUT_OIL:SUNFLOWER_OIL, ~ as.numeric(gsub("[^0-9//.]", "", .))))

perbco <- commodities %>%
  filter(Fecha == as.Date("2020-10-01")) %>%  # Asegurar que la fecha es Date
  select(COCONUT_OIL:SUNFLOWER_OIL)

indco <- commodities %>%
  filter(Fecha >= as.Date("2020-10-01") & Fecha <= as.Date("2024-12-01")) %>%
  mutate(across(COCONUT_OIL:SUNFLOWER_OIL, ~ (.x / perbco[[cur_column()]]) * 100, .names = "Indice_{.col}")) %>%
  select(Fecha, starts_with("Indice_"))

# Crear ndices del IPC de aceites
ipc_aceites <- ipc_aceites %>%
  mutate(across(Aceite_y_grasas:Aceite_puro_de_origen_vegetal, ~ as.numeric(gsub("[^0-9//.]", "", .))))

ipc_base <- ipc_aceites %>%
  filter(year(Fecha) == 2020 & month(Fecha) == 10) %>%
  slice(1)

ipc_aceites <- ipc_aceites %>%
  mutate(
    Indice_Aceite_y_grasas = (Aceite_y_grasas / ipc_base$Aceite_y_grasas) * 100,
    Indice_Aceites = (Aceites / ipc_base$Aceites) * 100,
    Indice_Aceite_de_soya = (Aceite_de_soya / ipc_base$Aceite_de_soya) * 100,
    Indice_Aceite_vegetal = (Aceite_puro_de_origen_vegetal / ipc_base$Aceite_puro_de_origen_vegetal) * 100
  )

# Combinar datos del IPC de aceites con los ndices de commodities
series <- left_join(ipc_aceites, indco, by = "Fecha")

# Calcular rezago (lag) de 1 mes en todos los ndices
series <- series %>%
  arrange(Fecha) %>%
  mutate(across(starts_with("Indice_"), ~ lag(.x, n = 1), .names = "Lag_{.col}"))

  theme_minimal()

# Definir una ventana mvil de 6 meses
window_size <- 6  

# Aplicar la correlacin mvil entre el IPC de Aceites y el Aceite de Soya
series <- series %>%
  mutate(
    Correlacion_Movil = rollapply(
      data = select(series, Indice_Aceites, Indice_SOYBEAN_OIL),
      width = window_size,
      FUN = function(x) cor(x[,1], x[,2], use = "pairwise.complete.obs"),
      by.column = FALSE,
      fill = NA
    )
  )

# Graficar la correlacin mvil
ggplot(series, aes(x = Fecha, y = Correlacion_Movil)) +
  geom_line(color = "blue", size = 1) +
  labs(title = "Correlacin Mvil (6 meses) entre IPC de Aceites y Aceite de Soya",

  theme_minimal()
